function getPoint(){
	$.ajax({
		url : "https://m.cjone.com/cjmmobile/myone/getPointInfo.do?callback=?"
		,dataType:"jsonp"
		,async: false
		,type: "POST"
		,success:function(data){
			$('#menuPoint').html("");
			if(!isEmpty(data)){
				var param = data.point;
				$('#menuPoint').append(numberWithCommas(param+'P'));
			} else {
				$('#menuPoint').append('0P');
			}
		}
		,error:function(data){
			console.log(data.msg);
		}
	});
}

/** 데이터 Null 체크 함수 **/
function isEmpty (str) {
    if(typeof str=='undefined' || str==null || str =='') return true;
    else return false;
}

/** 1000단위 쉼표 추가 함수 **/
function numberWithCommas (x) {
    var result = '';
    if(!isEmpty(x)) result = x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    return result;
}
